TYPE.js (DEV)

왜 만들었는가?

  • jQuery에서는 왜 Element만 chainning을 하는가?
  • function, array, string.. 내가 만든 객체도 chainning

어떻게 사용하는가?


type.isDom

var dom = type.isDom(document.getElementById('test'));
dom.first().first().text('aa').next().text('bb').next().text('cc').end();

type.isArray
var array = [5,2,1,6,2];
var a = type.isArray(array).sort().rest().rest().end(); //[2,5,6];

type.isFunction
function fn(){return Asdf.A.merge([this.name],arguments)};
var f = type.isFunction(fn).bind({name: 'kim'}).curry('aa','bb').end();
f(); //['kim', 'aa', 'bb']

/**
* 내가 만든 객체
* type.create(name, 타입체크함수, prototype);
* type.isName 으로 객체 생성.
*/
type.create('people',
    function (obj){ return obj.name && obj.age},
    {
        name: function(name){
            this.data.name = name;
            return this;
        },
        age: function(age){
            this.data.age = age;
            return this;
        }
    }
);

type.isPeople({name:'kim', age: '20'}).name('lee').age('25').end();
//{name:'lee', age: '25'}
